Lecture 5A - Dynamic Document with JavaScript

Change content with JavaScript

Step:

  1. Assign ID to HTML element.
  2. Get element: document.getElementById("id").
  3. Change: innerHTML for span/div value for input, src for img.

Useful Tags for Dynamic Content

<div>: Block container.

<span>: Inline container.

Example 1: Say Hi!

HTML code:

        Firstname: <input type="text" id="firstname1" /> <br>
        Lastname: <input type="text" id="lastname1" /> <br>
        <button onclick="sayHi1()" >Say Hi</button> <br>
        <span id="greeting"></span>

Function sayHi1() code:

        function sayHi1() {
        var firstname1 = document.getElementById("firstname1").value;
        var lastname1 = document.getElementById("lastname1").value;
        var message = "Hi " + firstname1 + " " + lastname1 + "!";
        document.getElementById("greeting").innerHTML = message;
        }

Example 3: Math Question

Enter 2 numbers, select operation, click "=" for result.

HTML code:

        <input type="text" id="input1" />
        <select id="operationSelect">
            <option value="add">+</option>
            <option value="subtract">-</option>
            <option value="multiply">x</option>
        </select>
        <input type="text" id="input2" />
        <button onclick="answer()">=</button>
        <input type="text" id="result" />

Function answer() code:

        function answer() {
            // Get the 1st number
            var inputField1 = document.getElementById("input1").value;
            var num1 = Number(inputField1);
            // Get 2nd number
            var inputField2 = document.getElementById("input2").value;
            var num2 = Number(inputField2);
            // Get the operation
            var operation = document.getElementById("operationSelect").value;
            // Calculate result
            var result;
            if(operation == "add") {
                result = num1 + num2;
            } else if(operation == "subtract") {
                result = num1 - num2;
            } else if(operation == "multiply") {
                result = num1 * num2;
            } else if(operation == "divide") {
                result = num1 / num2;
            }
            // Display the result
            document.getElementById("result").value = result;
        }

Example 4: Cat & Dog

Click "Cat" or "Dog" to show corresponding image and message.

HTML code:

        <button onclick="showAnimal('Meao meao meao!', 'cat.jpg')">Cat</button>
        <button onclick="showAnimal('Woof woof woof!', 'dog.jpg')">Dog</button>
        <br> <br>
        <span id="animalMessage"></span> <br>
        <img id="animalImage" src="" alt="Animal" style="max-width: 300px; display: none;">

Function showAnimal(message, imageFile) code:

        function showAnimal(message, imageFile) {
            document.getElementById("animalMessage").innerHTML = message;
            var img = document.getElementById("animalImage");
            img.src = imageFile;
            img.style.display = "block";
        }

Example 6: Counter Animation

HTML code:

Function code:

        var counter = 0;
        var counterSchedule;
        function startCounterAnimation() {
            counterSchedule = setInterval(showCounter, 1000);
        }
        function showCounter() {
            counter++;
            document.getElementById("counter").innerHTML = counter;
        }
        function stopCounterAnimation() {
            clearInterval(counterSchedule);
        }

Example 7: Dice Animation

HTML code:

Function rollDiceAnimation() code:

        var rollDiceCounter = 0;
        var rollDiceSchedule;
        function rollDiceAnimation() {
            rollDiceCounter = 0;
            rollDiceSchedule = setInterval(rollDice, 100);
        }
        function rollDice() {
            var diceValue = Math.floor(Math.random() * 6) + 1;
            var imageFile = "dice" + diceValue + ".png";
            document.getElementById("dice").src = imageFile;
            rollDiceCounter++;
            if (rollDiceCounter == 10) {
                clearInterval(rollDiceSchedule);
            }
        }
